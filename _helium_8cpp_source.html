271 static bool
272 _poll_get_result_handler(void * handler_ctx,
273  const char * key,
274  enum helium_config_type value_type,
275  void * value)
276 {
277  struct _poll_get_context * ctx = (struct _poll_get_context *)handler_ctx;
278  size_t len = ctx->dest_len;
279  void * src = value;
280 
281  if (strcmp(ctx->filter_key, key) != 0)
282  {
283  // Not the right key, keep going
284  return true;
285  }
286  else if (value_type == ctx->filter_type)
287  {
288  // Found and the right type
289  // Use the given value and the destination buffer size
290  ctx->status = config_status_POLL_FOUND;
291  }
292  else
293  {
294  // Found but not the right type, return an error
295  ctx->status = config_status_POLL_ERR_TYPE;
296  return false;
297  }
298 
299  switch (value_type)
300  {
301  case helium_config_bool:
302  case helium_config_f32:
303  case helium_config_i32:
304  case helium_config_str:
305  break;
306  case helium_config_null:
307  ctx->status = config_status_POLL_FOUND_NULL;
308  len = 0;
309  break;
310  }
311  memcpy(ctx->dest, src, len);
312  return false;
313 }
